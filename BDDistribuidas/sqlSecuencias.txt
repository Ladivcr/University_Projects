CREATE DATABASE Morelia;
CREATE DATABASE Patzcuaro;
USE Morelia;
CREATE TABLE Clientes (
    Nombre VARCHAR(100) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL,
    ApellidoP VARCHAR(100) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL,
    ApellidoM VARCHAR(100) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL,
    FeNacimiento DATE NOT NULL,
    RFC VARCHAR(13) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL PRIMARY KEY);


CREATE TABLE Direcciones(Id INT(11) AUTO_INCREMENT PRIMARY KEY,
    Calle VARCHAR(150) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL,
    Numero INT(11) NOT NULL,
    Colonia VARCHAR(150) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL,
    Estado VARCHAR(150) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL,
    CP CHAR(5) NOT NULL,
    RFC VARCHAR(13) CHARACTER SET utf8 COLLATE utf8_spanish_ci,
    FOREIGN KEY (RFC) REFERENCES Clientes(RFC));
